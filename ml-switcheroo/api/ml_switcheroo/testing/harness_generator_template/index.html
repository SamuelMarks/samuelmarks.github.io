def _make_jax_key(seed):
    "Attempts to create a JAX PRNGKey."
    try:
        import jax
        import jax.random
        return jax.random.PRNGKey(seed)
    except ImportError:
        return "mock_jax_key"

def _make_flax_rngs(seed):
    "Attempts to create a Flax NNX Rngs object."
    try:
        from flax import nnx
        return nnx.Rngs(seed)
    except ImportError:
        return "mock_flax_rngs"

def to_numpy(obj):
    try:
        import numpy as np
    except ImportError:
        return obj

    if hasattr(obj, "detach"): # Torch
        return obj.detach().cpu().numpy()
    if hasattr(obj, "__array__"): # JAX/Numpy
        return np.array(obj)
    if hasattr(obj, "numpy"): # TF
        return obj.numpy()
    if isinstance(obj, (list, tuple)):
         return [to_numpy(x) for x in obj]
    return obj

def load_module_from_path(name, path):
    spec = importlib.util.spec_from_file_location(name, path)
    if spec is None:
        print(f"‚ùå Could not load module {{name}} at {{path}}")
        return None
    mod = importlib.util.module_from_spec(spec)
    sys.modules[name] = mod
    try:
        spec.loader.exec_module(mod)
    except Exception as e:
        print(f"‚ùå Error during import of {{name}}: {{e}}")
        return None
    return mod

    print(f"üß™ Verifying: {{source_path}} vs {{target_path}}")

    mod_src = load_module_from_path("mod_src", source_path)
    mod_tgt = load_module_from_path("mod_tgt", target_path)

        try:
            sig = inspect.signature(src_func)

        # Adapt Inputs

                     # Numpy like

                if isinstance(a, (list, tuple)) and isinstance(b, (list, tuple)):
